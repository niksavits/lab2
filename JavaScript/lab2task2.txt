const numsToReverse = [123, 456, 789]; // Элементы для инициализации

function reverseNumber(num) {
    let reversed = 0;
    while (num > 0) {
        reversed = reversed * 10 + num % 10;
        num = Math.floor(num / 10);
    }
    return reversed;
}

const nums = numsToReverse.map(num => reverseNumber(num));

console.log(nums.join(' '));
