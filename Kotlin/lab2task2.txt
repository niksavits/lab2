import java.util.*

fun main() {
    val scanner = Scanner(System.`in`)
    val nums = ArrayList<Int>()
    print("Enter the quantity of numbers: ")
    val n = scanner.nextInt()
    for (i in 0 until n) {
        print("Element: ")
        var elem = scanner.nextInt()
        var revElem = 0
        while (elem > 0) {
            revElem = revElem * 10 + elem % 10
            elem /= 10
        }
        nums.add(revElem)
    }
    for (elems in nums) {
        print("$elems ")
    }
}
