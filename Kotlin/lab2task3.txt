import java.util.Scanner;
import java.util.ArrayList;

fun main() {
    val input = Scanner(System.`in`)
    val nums = ArrayList<Int>()
    
    print("Введите количество символов: ")
    val n = input.nextInt()
    
    for (i in 0 until n) {
        print("Элемент: ")
        var elem = input.nextInt()
        var revElem = 0
        
        while (elem > 0) {
            revElem = revElem * 10 + elem % 10
            elem /= 10
        }
        
        nums.add(revElem)
    }
    
    for (elem in nums) {
        print("$elem ")
    }

}
