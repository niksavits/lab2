puts "Введите количество символов:"
n = gets.chomp.to_i
nums = []

n.times do
    puts "Элемент:"
    elem = gets.chomp.to_i
    rev_elem = 0
    while elem > 0
        rev_elem = rev_elem * 10 + elem % 10
        elem = elem / 10
    end
    nums.push(rev_elem)
end

nums.each do |elem|
    print elem.to_s + " "
end
