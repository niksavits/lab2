glasnye = "yaoeiu"

s = input("Введите строку: ")

quantityGlas = 0

for char in s:
    if char.lower() in glasnye:
        quantityGlas += 1

soglas = len(s) - quantityGlas

if quantityGlas > soglas:
    print("Да")
elif quantityGlas == soglas:
    print("Одинаково")
else:
    print("Нет")
