nums = []
n = int(input("Введите количество символов: "))

for _ in range(n):
    elem = int(input("Элемент: "))
    rev_elem = 0
    while elem > 0:
        rev_elem = rev_elem * 10 + elem % 10
        elem = elem // 10
    nums.append(rev_elem)

for elems in nums:
    print(elems, end=" ")
