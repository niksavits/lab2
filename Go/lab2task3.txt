package main

import (
 "fmt"
)

func main() {
 var nums []int
 var n int
 fmt.Print("Введите количество символов: ")
 fmt.Scan(&n)
 for i := 0; i < n; i++ {
  var elem int
  fmt.Print("Элемент: ")
  fmt.Scan(&elem)
  revElem := 0
  for elem > 0 {
   revElem = revElem*10 + elem%10
   elem = elem / 10
  }
  nums = append(nums, revElem)
 }
 for _, elems := range nums {
  fmt.Print(elems, " ")
 }
}
