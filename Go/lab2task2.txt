package main

import (
 "fmt"
)

func main() {
 var nums []int
 var n int
 fmt.Println("Введите количество символов: ")
 fmt.Scan(&n)

 for i := 0; i < n; i++ {
  var elem, revElem int
  fmt.Println("Элемент: ")
  fmt.Scan(&elem)

  for elem > 0 {
   revElem = revElem*10 + elem%10
   elem = elem / 10
  }
  nums = append(nums, revElem)
 }

 for _, elems := range nums {
  fmt.Print(elems, " ")
 }
}
